// --- MAIN WRAPPER COMPONENT ---

const EmbryoMetrixWeb = () => {
  const [currentView, setCurrentView] = useState('login'); 
  const [assessmentStep, setAssessmentStep] = useState(0);
  const [selectedReport, setSelectedReport] = useState(null);
  const [isEditingProfile, setIsEditingProfile] = useState(false);

  // ASSESSMENT STATE
  const [assessmentData, setAssessmentData] = useState({
    patientId: '', patientName: '', dob: '', age: '',
    q1_day: '', q2_duration: '', q2_unit: 'Hours',
    q3_medium: '', q4_color: '', q5_ph: '', q6_clarity: '',
    photo: null, video: null, notes: ''
  });

  const [chatOpen, setChatOpen] = useState(false);
  const [chatMessage, setChatMessage] = useState('');
  const [chatHistory, setChatHistory] = useState([
    { role: 'ai', text: 'Hello! I am your AI assistant. How can I help with your embryo assessments today?' }
  ]);

  const handleSendMessage = () => {
    if (!chatMessage.trim()) return;
    const newHistory = [...chatHistory, { role: 'user', text: chatMessage }];
    setChatHistory(newHistory);
    // Mock response logic
    setTimeout(() => {
        let response = "I'm analyzing your data. Please wait...";
        if (chatMessage.toLowerCase().includes('viability')) response = "Viability is calculated using media markers and morphology.";
        if (chatMessage.toLowerCase().includes('score')) response = "High scores indicate strong metabolic activity.";
        setChatHistory([...newHistory, { role: 'ai', text: response }]);
    }, 1000);
    setChatMessage('');
  };

  // PROFILE STATE
  const [profile, setProfile] = useState({
    name: 'Dr. Amelia Chen', specialty: 'Reproductive Endocrinologist',
    experience: '10+ years', clinicName: 'Advanced Fertility Center',
    address: '123 Main Street, Anytown, USA', phone: '+1 (555) 123-4567',
    email: 'amelia.chen@clinic.com'
  });

  // MOCK DATA STORE
  const [reportsStore, setReportsStore] = useState([
    { id: "R-9021", patientName: "Sophia Carter", date: "24 Feb 2026", score: 95, age: 36, q1_day: 'D5', q6_clarity: 'Clear' },
    { id: "R-9018", patientName: "Marcus Reid", date: "22 Feb 2026", score: 82, age: 41, q1_day: 'D3', q6_clarity: 'Slightly turbid' },
    { id: "R-8995", patientName: "Elena Vance", date: "20 Feb 2026", score: 71, age: 29, q1_day: 'D6', q6_clarity: 'Turbid' },
  ]);

  // NAVIGATION HANDLERS
  const startNewAssessment = () => {
    setAssessmentData({ patientId: '', patientName: '', dob: '', age: '', q1_day: '', q2_duration: '', q2_unit: 'Hours', q3_medium: '', q4_color: '', q5_ph: '', q6_clarity: '', photo: null, video: null, notes: '' });
    setAssessmentStep(0);
    setCurrentView('assessment');
    setSelectedReport(null);
  };

  const openReportDetail = (report) => {
    setSelectedReport(report);
    setCurrentView('reportDetail');
  };

  const submitAssessment = () => {
    const newReport = {
      id: `R-${Math.floor(Math.random() * 10000)}`,
      patientName: assessmentData.patientName || "Anonymous",
      date: new Date().toLocaleDateString('en-GB', { day: 'numeric', month: 'short', year: 'numeric' }),
      score: Math.floor(Math.random() * 20) + 80,
      age: assessmentData.age || 30,
      q1_day: assessmentData.q1_day,
      q6_clarity: assessmentData.q6_clarity,
      notes: assessmentData.notes
    };
    setReportsStore([newReport, ...reportsStore]);
    openReportDetail(newReport);
  };

  // RENDER LOGIC
  if (currentView === 'login') return <LoginView onLogin={() => setCurrentView('dashboard')} />;

  return (
    <div className="min-h-screen bg-[#FBFCFE] text-gray-900 selection:bg-blue-100 font-sans">
      <div className="flex md:flex-row flex-col min-h-screen">
        
        {/* SIDE BAR */}
        <nav className="md:w-32 bg-white border-r border-gray-100 flex md:flex-col items-center py-10 px-6 gap-8 sticky top-0 md:h-screen z-50">
          <div onClick={() => setCurrentView('dashboard')} className="w-14 h-14 bg-blue-600 rounded-[1.2rem] flex items-center justify-center text-white font-black text-2xl shadow-2xl cursor-pointer">EM</div>
          <div className="flex md:flex-col gap-6">
             {[
               { id: 'dashboard', icon: Home },
               { id: 'reportsList', icon: FileText },
               { id: 'profile', icon: User }
             ].map(item => (
               <button key={item.id} onClick={() => { setCurrentView(item.id); setSelectedReport(null); }} className={`p-4 rounded-2xl transition-all ${currentView === item.id || (item.id === 'reportsList' && currentView === 'reportDetail') ? 'bg-blue-600 text-white shadow-2xl shadow-blue-200' : 'text-gray-300 hover:text-blue-600 hover:bg-blue-50'}`}>
                 <item.icon size={26} strokeWidth={3} />
               </button>
             ))}
             <button onClick={startNewAssessment} className="p-4 rounded-2xl text-blue-600 bg-blue-50 border-2 border-blue-100 hover:bg-blue-600 hover:text-white transition-all">
                <Plus size={26} strokeWidth={3} />
             </button>
          </div>
          <button onClick={() => setCurrentView('login')} className="mt-auto p-4 text-gray-200 hover:text-red-500 hover:bg-red-50 rounded-2xl transition-all group">
            <LogOut size={26} strokeWidth={3} />
          </button>
        </nav>

        {/* CONTENT */}
        <main className="flex-1 p-8 md:p-20 max-w-7xl mx-auto w-full md:pb-20 pb-40">
           {currentView === 'dashboard' && <DashboardView profile={profile} onStartAssessment={startNewAssessment} onViewReport={openReportDetail} recentReports={reportsStore.slice(0, 3)} />}
           {currentView === 'reportsList' && <ReportsListView reports={reportsStore} onViewReport={openReportDetail} onNewAssessment={startNewAssessment} />}
           {currentView === 'reportDetail' && <ResultsView report={selectedReport} onBackToReports={() => setCurrentView('reportsList')} />}
           {currentView === 'profile' && <ProfileView profile={profile} setProfile={setProfile} isEditing={isEditingProfile} setIsEditing={setIsEditingProfile} />}
           {currentView === 'assessment' && (
             <AssessmentView 
               step={assessmentStep} 
               data={assessmentData} 
               setData={setAssessmentData} 
                onNext={() => assessmentStep < 9 ? setAssessmentStep(assessmentStep + 1) : submitAssessment()} 
                onPrev={() => setAssessmentStep(Math.max(0, assessmentStep - 1))}
              />
           )}
        </main>

        {/* AI CHATBOT FLOAT */}
        <div className="fixed bottom-6 right-6 z-[100] md:flex hidden flex-col items-end gap-4">
            {chatOpen && (
                <div className="w-96 h-[500px] bg-white rounded-[2.5rem] shadow-2xl border border-gray-100 flex flex-col overflow-hidden animate-in slide-in-from-bottom-5">
                    <div className="p-6 bg-blue-600 text-white flex justify-between items-center">
                        <div className="flex items-center gap-3">
                            <Zap size={20} />
                            <span className="font-black uppercase tracking-widest text-xs">AI Assistant</span>
                        </div>
                        <button onClick={() => setChatOpen(false)}><X size={20} /></button>
                    </div>
                    <div className="flex-1 p-6 overflow-y-auto space-y-4">
                        {chatHistory.map((h, i) => (
                            <div key={i} className={`flex ${h.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                                <div className={`max-w-[80%] p-4 rounded-2xl font-bold text-sm ${h.role === 'user' ? 'bg-blue-600 text-white' : 'bg-gray-100 text-gray-700'}`}>
                                    {h.text}
                                </div>
                            </div>
                        ))}
                    </div>
                    <div className="p-6 border-t border-gray-50 flex gap-2">
                        <input value={chatMessage} onChange={e => setChatMessage(e.target.value)} onKeyPress={e => e.key === 'Enter' && handleSendMessage()} className="flex-1 bg-gray-50 p-4 rounded-xl font-bold text-sm outline-none" placeholder="Ask AI..." />
                        <button onClick={handleSendMessage} className="bg-blue-600 text-white p-4 rounded-xl"><ArrowRight size={20} /></button>
                    </div>
                </div>
            )}
            <button onClick={() => setChatOpen(!chatOpen)} className="w-20 h-20 bg-[#1e1b4b] text-white rounded-3xl shadow-2xl flex items-center justify-center hover:scale-110 transition-all group">
                <Zap size={32} className="group-hover:text-blue-400 transition-colors" />
            </button>
        </div>

        {/* MOBILE BOTTOM NAV */}
        <div className="md:hidden fixed bottom-6 left-6 right-6 bg-white/90 backdrop-blur-xl border border-gray-100 rounded-[2.5rem] py-3 px-6 shadow-2xl z-[100] flex justify-between items-center">
           <button onClick={() => setCurrentView('dashboard')} className={`p-3 rounded-2xl ${currentView === 'dashboard' ? 'text-blue-600 bg-blue-50' : 'text-gray-400'}`}><Home size={24} strokeWidth={3}/></button>
           <button onClick={startNewAssessment} className="p-5 bg-blue-600 text-white rounded-[1.5rem] shadow-xl active:scale-90 transition-all -translate-y-8 border-8 border-[#FBFCFE]"><PlusSquare size={28} strokeWidth={3}/></button>
           <button onClick={() => setCurrentView('reportsList')} className={`p-3 rounded-2xl ${currentView === 'reportsList' || currentView === 'reportDetail' ? 'text-blue-600 bg-blue-50' : 'text-gray-400'}`}><FileText size={24} strokeWidth={3}/></button>
        </div>
      </div>
    </div>
  );
};

export default EmbryoMetrixWeb;
